    def prime_factorization(x):
        factors = []
        while x % 2 == 0:
            factors.append(2)
            x = x // 2
        i = 3
        while i * i <= x:
            while x % i == 0:
                factors.append(i)
                x = x // i
            i += 2
        if x > 1:
            factors.append(x)
        return factors

    io = remote('contest.ctf.nefu.edu.cn', 33772)

    while True:

            response = io.recvline(timeout=1).decode().strip()
